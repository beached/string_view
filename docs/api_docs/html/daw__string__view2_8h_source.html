916 template<string_view_bounds_type BL, string_view_bounds_type BR>
917 [[nodiscard]] static constexpr int
918 compare( basic_string_view<CharT, BL> lhs,
919 basic_string_view<CharT, BR> rhs ) {
920 auto const str_compare = []( CharT const *p0, CharT const *p1,
921 size_type len ) {
922 auto const last = p0 + len;
923 while( p0 != last ) {
924 if( *p0 != *p1 ) {
925 if( *p0 < *p1 ) {
926 return -1;
927 }
928 return 1;
929 }
930 ++p0;
931 ++p1;
932 }
933 return 0;
934 };
935
936 auto cmp = str_compare( lhs.data( ), rhs.data( ),
937 ( std::min )( lhs.size( ), rhs.size( ) ) );
938 if( cmp == 0 ) {
939 if( lhs.size( ) < rhs.size( ) ) {
940 return -1;
941 }
942 if( rhs.size( ) < lhs.size( ) ) {
943 return 1;
944 }
945 }
946 return cmp;
947 }
