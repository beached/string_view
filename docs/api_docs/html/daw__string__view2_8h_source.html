931 template<string_view_bounds_type BL, std::ptrdiff_t ExL,
932 string_view_bounds_type BR, std::ptrdiff_t ExR>
933 [[nodiscard]] static constexpr int
934 compare( basic_string_view<CharT, BL, ExL> lhs,
935 basic_string_view<CharT, BR, ExR> rhs ) {
936 auto const str_compare = []( CharT const *p0, CharT const *p1,
937 size_type len ) {
938 auto const last = p0 + len;
939 while( p0 != last ) {
940 if( *p0 != *p1 ) {
941 if( *p0 < *p1 ) {
942 return -1;
943 }
944 return 1;
945 }
946 ++p0;
947 ++p1;
948 }
949 return 0;
950 };
951
952 auto cmp = str_compare( lhs.data( ), rhs.data( ),
953 ( std::min )( lhs.size( ), rhs.size( ) ) );
954 if( cmp == 0 ) {
955 if( lhs.size( ) < rhs.size( ) ) {
956 return -1;
957 }
958 if( rhs.size( ) < lhs.size( ) ) {
959 return 1;
960 }
961 }
962 return cmp;
963 }
